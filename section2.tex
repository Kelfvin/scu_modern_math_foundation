\subsection{Lagrange插值}

$p(x) = \sum_{i=0}^n y_{i}l_{i}({x}) \qquad
l_{i}(x) = \prod_{j=0,i \neq j}^n \frac{x-x_{j}}{x_{i}-x_{j}}$

$R_{n}(x) = f(x) - L_{n}(x)= \frac{f^{(n-1)}(\xi)}{(n+1)!} \omega(x)$

\subsection{NewTon插值}

差商：

\begin{align*}
f[x_{i},x_{j}] &= \frac{f_{j}-f_{i}}{x_{j}-x_{i}} &&\text{一阶差商} \\
f[x_{i},x_{j},x_{k}] &= \frac{f[x_{k},x_{j}]-f[x_{j},x_{i}]}{x_{k}-x_{i}} &&\text{二阶差商} \\
f[x_{0},x_{1},x_{2},\dots,x_{k}] &= \frac{f[x_{1},x_{2},\dots ,x_{k}]-f[x_{0},x_{1},\dots,x_{k-1}]}{x_{k}-x_{0}} &&\text{k阶差商}
\end{align*} 


差商的性质：

\begin{itemize}
    \item 差商可以表示为函数值的线性组合
    \begin{itemize}
        \item $f[x_{0},x_{1},\dots,x_{n}]=\sum_{j=0}^n \frac{f(x_{j})}{\omega'(x_{j})}$
        \item 其中 $\omega(x)=(x-x_{0})(x-x_{1})\dots(x-x_{n})$
    \end{itemize}

    \item 对称性：差商与节点的排列顺序无关
    \item $f[x_{0},\dots,x_{n}]=\frac{f^{(n)}(\xi)}{n!},(\xi \in (a,b))$
    \begin{itemize}
        \item 可用于快速计算一些差商值，如 $f(x)=x^2$，求 $f[1,2,3,4]$
    \end{itemize}

\end{itemize}

NewTon 插值公式：

$
   N_n(x) = f(x_0)
    + f[x_0, x_1](x - x_0)
    + \cdots
    + f[x_0, \dots, x_n](x - x_0)\cdots(x - x_{n-1})
$

$
    R_{n}(x) = f[x,x_{0},\dots,x_{n}]\omega(x)
$
